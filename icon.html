Icons

Install depencies

to use icons install iconify package && iconify bi icons set in dev depencies

npm i -D @iconify/tailwind
npm i -D @iconify-json/bi

import icons plugin to (tailwind.config.js) and add it to the plugins

import { addIconSelectors } from "@iconify/tailwind";
/** @type {import('tailwindcss').Config} */
export default {
    plugins: [
    ...
        addIconSelectors({
            prefixes: [
                'bi',
            ],
            iconSelector: ".{prefix}-{name}",
            maskSelector: ".icon",
        }),
    ], 
}
bi-house-fill
bi-newspaper
bi-palette-fill
bi-info
bi-shield-fill-check
